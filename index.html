<link rel="stylesheet" href="./slide.css">

Default

new Slide('#slide', {
    imgs: [
        './dist/imgs/1.gif',
        './dist/imgs/2.gif',
        './dist/imgs/3.gif',
    ]
})

Animation: slideLeft

new Slide('#slide', {
    imgs: [
        './dist/imgs/1.gif',
        './dist/imgs/2.gif',
        './dist/imgs/3.gif',
    ],
    animation: 'slideLeft'
})

Animation: fade

new Slide('#slide', {

        './dist/imgs/1.gif',

Autoswitch

new Slide('#slide', {
    imgs: [
        './dist/imgs/1.gif',
        './dist/imgs/2.gif',
        './dist/imgs/3.gif',
    ],
    autoswitch: {
        open: true,
        delay: 3000
    }
})
